function generarCotizacion() { const { jsPDF } = window.jspdf; const doc = new jsPDF(); // Datos capturados const emp = document.getElementById('empresa').value || "Cliente Corporativo"; const m2 = parseInt(document.getElementById('metros').value) || 0; const tIva = parseFloat(document.getElementById('zona').value); const flujo = document.getElementById('flujo').value; const incluyeInsumos = document.getElementById('insumos').value; // Lógica de cálculo MESAN let ratio = flujo === 'hospital' ? 120 : 250; let personal = Math.ceil(m2 / ratio); let subtotal = (personal * 18500) * 1.30; if(incluyeInsumos === 'si') subtotal += (m2 * 15); const iva = subtotal * tIva; const total = subtotal + iva; // --- DISEÑO ESTILO "PULIZIA" --- // 1. Encabezado doc.setFont("helvetica", "bold"); doc.setFontSize(22); doc.setTextColor(0, 45, 90); // Azul profesional doc.text("MESAN Ω", 20, 25); doc.setFontSize(8); doc.text("SERVICIOS INDUSTRIALES", 20, 30); doc.setFontSize(10); doc.setTextColor(100); doc.text("Cotización No. 2026-Ω", 150, 25); doc.text(`Mexicali, B.C. a ${new Date().toLocaleDateString()}`, 150, 30); // 2. Destinatario doc.setFontSize(11); doc.setTextColor(0); doc.text(`At'n:`, 20, 45); doc.text(`Empresa: ${emp.toUpperCase()}`, 20, 50); doc.text(`Asunto: Propuesta de servicio de mantenimiento y limpieza`, 20, 55); // 3. Detalle del Servicio doc.text("Detalle del servicio:", 20, 70); doc.setFont("helvetica", "normal"); doc.text(`• Personal asignado: ${personal} personas capacitadas.`, 25, 80); doc.text(`• Horario: Lunes a Sábado (Turnos según flujo ${flujo}).`, 25, 87); doc.text(`• Insumos: ${incluyeInsumos === 'si' ? 'INCLUYE materiales y consumibles.' : 'NO incluye insumos de limpieza.'}`, 25, 94); // 4. Costos (Cuadro resaltado) doc.setFont("helvetica", "bold"); doc.text("Costo del servicio:", 20, 110); doc.setFont("helvetica", "normal"); doc.text(`• Costo del servicio por mes: $${subtotal.toLocaleString()} M.N. más ${tIva * 100}% de IVA.`, 25, 120); // Desglose Final doc.setDrawColor(200); doc.line(120, 130, 190, 130); doc.text(`Subtotal:`, 120, 138); doc.text(`$${subtotal.toLocaleString()}`, 165, 138); doc.text(`IVA (${tIva * 100}%):`, 120, 145); doc.text(`$${iva.toLocaleString()}`, 165, 145); doc.setFont("helvetica", "bold"); doc.text(`TOTAL MENSUAL:`, 120, 155); doc.text(`$${total.toLocaleString()}`, 165, 155); // 5. Condiciones y Vigencia doc.setFontSize(10); doc.text("Condiciones adicionales:", 20, 175); doc.setFont("helvetica", "normal"); doc.setFontSize(9); doc.text("• Los pagos se realizan vía transferencia electrónica.", 25, 185); doc.text("• Enviar Constancia de Situación Fiscal actualizada para facturación.", 25, 192); doc.setFont("helvetica", "bold"); const fechaVence = new Date(); fechaVence.setDate(fechaVence.getDate() + 15); doc.text(`• Vigencia de esta cotización: ${fechaVence.toLocaleDateString()}.`, 25, 199); // 6. Cierre y Firma (Imagen cargada) doc.text("Atentamente,", 20, 220); // Aquí es donde se inserta tu firma (usando el formato de imagen que enviaste) // Nota: Para producción, la imagen debe convertirse a Base64 // doc.addImage(imgFirmaBase64, 'PNG', 20, 230, 50, 30); doc.text("__________________________", 20, 250); doc.text("Lic. Manuel Sánchez", 20, 255); doc.text("MESAN SERVICIOS Ω", 20, 260); doc.save(`Cotizacion_MESAN_${emp}.pdf`); }
